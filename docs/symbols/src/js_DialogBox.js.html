  6 /**
  7  * @class Отвечает за работу с диалоговым окном.
  8  * @constructor
  9  * @name DialogBox
 10  * @param {ErrorListManager} ErrorListManager Класс, отвечающий за работу со списком ошибок

 15 };

 27 /**
 28  * Добавляет текст к элементу textarea диалогового окна.
 29  *
 30  * @param {string} text Текст для добавления
 31  */
 32 DialogBox.prototype.addTextToMessage = function (text) {
 33     var messageTextarea = $('#dialog-message-text');
 34     messageTextarea.text(messageTextarea.val(messageTextarea.val() + text));
 35 };

 38 /**
 39  * Возвращает тип ошибки (раздел правил), который был выбран пользователем.
 40  *
 41  * @return {string} Значение text тега option в ввиде строки.
 42  */
 43 DialogBox.prototype.getErrorType = function() {
 44     return $('#type-of-error option:selected').text();
 45 };
 46 /**
 47  * Добавляет событие onChange для элемента c id='type-of-error' диалогового окна.
 48  * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
 49  * зависят от его содержания.
 50  */
 51 DialogBox.prototype.addEventTypeOfError_Change = function () {
 52     var dialogBox = this;
 53 
 54     var _EventHandler = function () {
 55         var errorType = dialogBox.getErrorType(),
 56             subtypeOfErrorSelect = $('#subtype-of-error'),
 57             messageToAppend = '\n';
 58 
 59         subtypeOfErrorSelect.empty();
 60         messageToAppend += dialogBox.ErrorListManager.getErrorMessageText(errorType);
 61         if (dialogBox.ErrorListManager.isErrorHasSubTypes(errorType)) {
 62             dialogBox.appendErrorSubTypesList(errorType, subtypeOfErrorSelect);
 63             subtypeOfErrorSelect.show();
 64         }
 65 
 66         dialogBox.addTextToMessage(messageToAppend);
 67     }
 68 
 69     $('#type-of-error').change(_EventHandler);
 70     $('#type-of-error').dblclick(_EventHandler);
 71 };
 72 
 73 /**
 74  * Добавляет событие onChange для элемента c id='subtype-of-error' диалогового окна.
 75  * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
 76  * зависят от его содержания.
 77  */
 78 DialogBox.prototype.addEventSubTypeOfError_Change = function () {
 79     var dialogBox = this;
 80 
 81     var _EventHandler = function () {
 82         var messageToAppend = '',
 83             errorSubType = dialogBox.getErrorSubType();
 84 
 85         messageToAppend += dialogBox.ErrorListManager.getErrorMessageText(errorSubType);
 86         dialogBox.addTextToMessage(messageToAppend);
 87     }
 88 
 89     $('#subtype-of-error').change(_EventHandler);
 90     $('#subtype-of-error').dblclick(_EventHandler);
 91 };

 93 /**
 94  * Добавляет слой с диалоговым окном к элементу body текущей страницы.
 95  * Добавляет события для контролов внутри этого слоя.
 96  */
 97 DialogBox.prototype.appendDialog = function () {
 98     var dialogBox = this;
 99     $('body').append('<div id="habracorrect-dialog" title="Сообщение об ошибке" style=' + dialogBox.DIALOG_STYLE + ' hidden>' +
100         '<p id="dialog-author-name"></p>' +
101         '<p id="dialog-message-title"><b>Заголовок:</b> </p>' +
102         '<p><b>Текст сообщения:</b></p>' +
103         '<textarea id="dialog-message-text" cols="60" rows="8" maxlength="1000" required></textarea>' +
104         '<select id="type-of-error">' +
105         '</select>' +
106         '</br>' +
107         '<select id="subtype-of-error" style="width : 200px" hidden></select>' +
108         '</div>');
109 
110     dialogBox.addTypeOfErrorOptions();
111     dialogBox.addEventTypeOfError_Change();
112     dialogBox.addEventSubTypeOfError_Change();
113 };

178 /**
179  * Добавляет к заданному контролу для перечисления ошибок
180  * option для каждого типа ошибки.
181  */
182 DialogBox.prototype.addTypeOfErrorOptions = function () {
183     var dialogBox = this;
184     var getTypeOfErrorSelect = function () {
185         return $('#type-of-error');
186     };
187 
188     var typeOfErrorSelect = getTypeOfErrorSelect();
189     if (dialogBox.ErrorListManager.errorDict) {
190         $.each(dialogBox.ErrorListManager.errorDict, function() {
191             typeOfErrorSelect.append(new Option(this.title, 0));
192         });
193     }
194 };
195 
196 /**
197  * Добавляет элементы option, которые соответствуют
198  * подтипам определенной ошибки, к заданному select
199  *
200  * @param {string} errorType Тип ошибки
201  * @param {jQuery object} subtypeOfErrorSelect Select для подтипов ошибок
202  */
203 DialogBox.prototype.appendErrorSubTypesList = function (errorType, subtypeOfErrorSelect) {
204     var dialogBox = this;
205     $.each(dialogBox.ErrorListManager.errorDict, function() {
206         if (this.title === errorType) {
207             $.each(this.children, function() {
208                 subtypeOfErrorSelect.append(new Option(this.title, 0));
209             });
210         }
211     });
212 };

214 /**
215  * Показывает диалоговое окно для отправки сообщения автору статьи.
216  * Добавляет события для контролов диалогового окна.
217  *
218  * @param {string} name Имя автора статьи
219  * @param {string} title Название статьи
220  */
221 DialogBox.prototype.showDialog = function (name, title, sendMessage) {
222     var dialogBox = this;
223     $('#habracorrect-dialog').attr('hidden', 'true');
224     $('#habracorrect-dialog').dialog({
225         width: 500
226     }, {
227         height: 360
228     }, {
229         buttons: {
230             "Опечатка!": function () {
231                 dialogBox.addTextToMessage("\nОпечатка!");
232             },
233             "Отправить": function () {
234                 sendMessage(name, title, dialogBox.getMessageText());
235                 $(this).dialog("close");
236             },
237             "Отменить": function () {
238                 $(this).dialog("close");
239             }
240         }
241     });
242 
243     $('.ui-dialog-buttonpane').find('button:contains("Опечатка!")').css("margin-right", "180px");
244 };
