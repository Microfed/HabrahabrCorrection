  6 /**

 29 /**
 30  * Добавляет текст к элементу textarea диалогового окна.
 31  *
 32  * @param {string} text Текст для добавления
 33  */
 34 DialogBox.prototype.addTextToMessage = function (text) {
 35     'use strict';
 36     var messageTextarea = $('#dialog-message-text');
 37     messageTextarea.text(messageTextarea.val(messageTextarea.val() + text));
 38 };

 41 /**
 42  * Возвращает тип ошибки (раздел правил), который был выбран пользователем.

 49 };
 50 /**
 51  * Добавляет событие onChange для элемента c id='type-of-error' диалогового окна.
 52  * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
 53  * зависят от его содержания.
 54  */
 55 DialogBox.prototype.addEventTypeOfError_Change = function () {
 56     'use strict';
 57     var dialogBox = this,
 58         EventHandler = function () {
 59             var errorType = dialogBox.getErrorType(),
 60                 subtypeOfErrorSelect = $('#subtype-of-error'),
 61                 messageToAppend = '\n';
 62 
 63             subtypeOfErrorSelect.empty();
 64             messageToAppend += dialogBox.ErrorListManager.getErrorMessageText(errorType);
 65             if (dialogBox.ErrorListManager.isErrorHasSubTypes(errorType)) {
 66                 dialogBox.appendErrorSubTypesList(errorType, subtypeOfErrorSelect);
 67                 subtypeOfErrorSelect.show();
 68             }
 69 
 70             dialogBox.addTextToMessage(messageToAppend);
 71         };
 72 
 73     $('#type-of-error').change(EventHandler);
 74     $('#type-of-error').dblclick(EventHandler);
 75 };
 76 
 77 /**
 78  * Добавляет событие onChange для элемента c id='subtype-of-error' диалогового окна.
 79  * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
 80  * зависят от его содержания.
 81  */
 82 DialogBox.prototype.addEventSubTypeOfError_Change = function () {
 83     'use strict';
 84     var dialogBox = this,
 85         EventHandler = function () {
 86             var messageToAppend = '',
 87                 errorSubType = dialogBox.getErrorSubType();
 88 
 89             messageToAppend += dialogBox.ErrorListManager.getErrorMessageText(errorSubType);
 90             dialogBox.addTextToMessage(messageToAppend);
 91         };
 92 
 93     $('#subtype-of-error').change(EventHandler);
 94     $('#subtype-of-error').dblclick(EventHandler);
 95 };

 97 /**
 98  * Добавляет слой с диалоговым окном к элементу body текущей страницы.
 99  * Добавляет события для контролов внутри этого слоя.
100  */
101 DialogBox.prototype.appendDialog = function () {
102     'use strict';
103     var dialogBox = this;
104     $('body').append('<div id="habracorrect-dialog" title="Сообщение об ошибке" style=' + dialogBox.DIALOG_STYLE + ' hidden>' +
105         '<p id="dialog-author-name"></p>' +
106         '<p id="dialog-message-title"><b>Заголовок:</b> </p>' +
107         '<p><b>Текст сообщения:</b></p>' +
108         '<textarea id="dialog-message-text" cols="60" rows="8" maxlength="1000" required></textarea>' +
109         '<select id="type-of-error">' +
110         '</select>' +
111         '</br>' +
112         '<select id="subtype-of-error" style="width : 200px" hidden></select>' +
113         '</div>');
114 
115     dialogBox.addTypeOfErrorOptions();
116     dialogBox.addEventTypeOfError_Change();
117     dialogBox.addEventSubTypeOfError_Change();
118 };

189 /**
190  * Добавляет к заданному контролу для перечисления ошибок
191  * option для каждого типа ошибки.
192  */
193 DialogBox.prototype.addTypeOfErrorOptions = function () {
194     'use strict';
195     var dialogBox = this,
196         getTypeOfErrorSelect = function () {
197             return $('#type-of-error');
198         },
199         typeOfErrorSelect = getTypeOfErrorSelect();
200 
201     if (dialogBox.ErrorListManager.errorDict) {
202         $.each(dialogBox.ErrorListManager.errorDict, function () {
203             typeOfErrorSelect.append(new Option(this.title, 0));
204         });
205     }
206 };
207 
208 /**
209  * Добавляет элементы option, которые соответствуют
210  * подтипам определенной ошибки, к заданному select
211  *
212  * @param {string} errorType Тип ошибки
213  * @param {jQuery object} subtypeOfErrorSelect Select для подтипов ошибок
214  */
215 DialogBox.prototype.appendErrorSubTypesList = function (errorType, subtypeOfErrorSelect) {
216     'use strict';
217     var dialogBox = this;
218     $.each(dialogBox.ErrorListManager.errorDict, function () {
219         if (this.title === errorType) {
220             $.each(this.children, function () {
221                 subtypeOfErrorSelect.append(new Option(this.title, 0));
222             });
223         }
224     });
225 };

227 /**
228  * Показывает диалоговое окно для отправки сообщения автору статьи.
229  * Добавляет события для контролов диалогового окна.
230  *
231  * @param {string} name Имя автора статьи
232  * @param {string} title Название статьи
233  */
234 DialogBox.prototype.showDialog = function (name, title, sendMessage) {
235     'use strict';
236     var dialogBox = this;
237     $('#habracorrect-dialog').attr('hidden', 'true');
238     $('#habracorrect-dialog').dialog({
239         width: 500
240     }, {
241         height: 360
242     }, {
243         buttons: {
244             "Опечатка!": function () {
245                 dialogBox.addTextToMessage("\nОпечатка!");
246             },
247             "Отправить": function () {
248                 sendMessage(name, title, dialogBox.getMessageText());
249                 $(this).dialog("close");
250             },
251             "Отменить": function () {
252                 $(this).dialog("close");
253             }
254         }
255     });
256 
257     $('.ui-dialog-buttonpane').find('button:contains("Опечатка!")').css("margin-right", "180px");
258 };
