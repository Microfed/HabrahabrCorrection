  6 /**Текст рекламы*/
  7 var AD_TEXT = '\n\n<sub>With help <a href="https://chrome.google.com/webstore/detail/kcdmenmdkfpfbdilcpfehcnahhkjfipe">HabraCorrection</a>.</sub>';
  8 
  9 /*
 10  Добавляет в сообщение информацию о расширении в качестве
 11  рекламы, если пользователь установил соответствующую опцию.
 12 
 13  @return {string} Возвращает строку, содержащую рекламный текст.
 14  */
 15 function getAdText() {
 16     'use strict';
 17     var isAdvtAttachToMessage = localStorage["isAdvtAttachToMessage"];
 18 
 19     if (isAdvtAttachToMessage === 'true') {
 20         return AD_TEXT;
 21     } else {
 22         return '';
 23     }
 24 }
 25 
 26 /**
 27  * Возвращает готовые для отправки данные.
 28  *
 29  * @param name Ник получателя
 30  * @param title Заголовок сообщения
 31  * @param text Текст сообщения
 32  * @return {string} Возвращает строку с подготовленными для отправки данными
 33  */
 34 function getData(name, title, text) {
 35     'use strict';
 36     return "message[recipients]=" + encodeURIComponent(name) + "&message[title]=" + encodeURIComponent(title) +
 37         "&message[text]=" + encodeURIComponent(text) + encodeURIComponent(getAdText());
 38 }
 39 
 40 chrome.extension.onRequest.addListener(
 41     function (request, sender, sendResponse) {
 42         'use strict';
 43         var name, title, text, data, errorList;
 44         if (request.to_send_data === true) {
 45             name = request.send_author;
 46             title = request.send_title;
 47             text = request.send_text;
 48             data = getData(name, title, text);
 49             sendResponse({send_data: data});
 50         }
 51         if (request.get_errorList) {
 52             errorList = JSON.parse(localStorage['errorList']);
 53             sendResponse({data: errorList});
 54         }
 55     }
 56 );
