  6 /**Текст рекламы*/
  7 var AD_TEXT = '\n\n<sub>With help <a href="https://chrome.google.com/webstore/detail/kcdmenmdkfpfbdilcpfehcnahhkjfipe">HabraCorrection</a>.</sub>';
  8 
  9 /*
 10  Добавляет в сообщение информацию о расширении в качестве
 11  рекламы, если пользователь установил соответствующую опцию.
 12 
 13  @return {string} Возвращает строку, содержащую рекламный текст.
 14  */
 15 function getAdText() {
 16     'use strict';
 17     var isAdvtAttachToMessage = localStorage["isAdvtAttachToMessage"];
 18 
 19     if (isAdvtAttachToMessage === 'true') {
 20         return AD_TEXT;
 21     } else {
 22         return '';
 23     }
 24 }
 25 
 26 /**
 27  * Возващает настройки из localStorage
 28  * @param {string} name Название опции
 29  */
 30 function getSettings(name) {
 31     'use strict';
 32     var value = '';
 33 
 34     if (name !== undefined) {
 35         if (localStorage[name] !== undefined) {
 36             value = localStorage[name];
 37         }
 38     }
 39 
 40     return value;
 41 }
 42 
 43 /**
 44  * Возвращает готовые для отправки данные.
 45  *
 46  * @param name Ник получателя
 47  * @param title Заголовок сообщения
 48  * @param text Текст сообщения
 49  * @return {string} Возвращает строку с подготовленными для отправки данными
 50  */
 51 function getData(name, title, text) {
 52     'use strict';
 53     return "message[recipients]=" + encodeURIComponent(name) + "&message[title]=" + encodeURIComponent(title) +
 54         "&message[text]=" + encodeURIComponent(text) + encodeURIComponent(getAdText());
 55 }
 56 
 57 chrome.extension.onRequest.addListener(
 58     function (request, sender, sendResponse) {
 59         'use strict';
 60         var name, title, text, data, errorList;
 61         if (request.to_send_data === true) {
 62             name = request.send_author;
 63             title = request.send_title;
 64             text = request.send_text;
 65             data = getData(name, title, text);
 66             sendResponse({send_data: data});
 67         }
 68         if (request.get_errorList) {
 69             errorList = JSON.parse(localStorage['errorList']);
 70             sendResponse({data: errorList});
 71         }
 72         if (request.get_silentMode) {
 73             sendResponse({silent_mode: getSettings('silentMode')});
 74         }
 75     }
 76 );
