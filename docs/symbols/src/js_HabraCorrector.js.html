  6 /**
  7  * Адресс скрипта отправки личных сообщений.
  8  */
  9 var URL_SEND_MESSAGE = "http://habrahabr.ru/ajax/messages/add/";
 10 
 11 /**
 12  * @class Отвечает за работу расширения.
 13  * @constructor
 14  * @name HabraCorrector
 15  */
 16 var HabraCorrector = function() {
 17     this.habraPage = new HabraPage();
 18     this.errorListManager;
 19     this.dialogBox;
 20 };
 21 
 22 /**
 23  * Загружает список ошибок из JSON-файла. После загрузки добавляет диалоговое окно
 24  * в DOM страницы.
 25  */
 26 HabraCorrector.prototype.loadErrorListAndAppendDialogBox = function() {
 27     var habraCorrector = this;
 28     $.getJSON(chrome.extension.getURL('/error-list.json'), function(data) {
 29         habraCorrector.errorListManager = new ErrorListManager(data);
 30         habraCorrector.dialogBox = new DialogBox(habraCorrector.errorListManager);
 31         habraCorrector.dialogBox.appendStyle();
 32         habraCorrector.dialogBox.appendDialog();
 33     });
 34 };
 35 
 36 /**
 37  * Инициализирует главный класс контент-скрипта:
 38  * - добавляет событие для горячих клавиш;
 39  * - загружает список ошибок;
 40  * - добавляет к DOM страницы диалоговое окно.
 41  */
 42 HabraCorrector.prototype.Init = function() {
 43     if (this.habraPage.isCurrentUrlCorrect()) {
 44         this.loadErrorListAndAppendDialogBox();
 45     }
 46     this.addHotKey();
 47 };
 48 
 49 /**
 50  * Проверяет ответ севера. Если ответ содержит информацию
 51  * об ошибке, сообщает эту информацию пользователю
 52  * посредством alert()
 53  *
 54  * @param response Ответ на запрос к серверу
 55  */
 56 HabraCorrector.prototype.alertWhenFail = function(response) {
 57     var resp = $(response).find('error');
 58     if (resp[0] !== undefined) {
 59         alert(resp[0].textContent);
 60     }
 61 };
 62 
 63 /**
 64  * Отправляет данные на сервер post-методом
 65  *
 66  * @param data Данные
 67  */
 68 HabraCorrector.prototype.sendDataPost = function(data) {
 69     var habraCorrector = this;
 70     $.post(URL_SEND_MESSAGE, data, function (response) {
 71         habraCorrector.alertWhenFail(response);
 72     });
 73 };
 74 
 75 /**
 76  * Отправляет сообщение фоновой странице расширения, отправляя данные для формирования сообщения
 77  * об ошибке; принимает ответ и шлет сформированное сообщение автору статьи. Для успешнйо отправки
 78  * сообщения требуется быть аутентифицированным пользователем.
 79  *
 80  * @param name Имя автора статьи
 81  * @param title Название статьи
 82  * @param messageText Текст сообщения
 83  */
 84 HabraCorrector.prototype.sendRequestToBackgroundPage = function(name, title, messageText) {
 85     chrome.extension.sendRequest({to_send_data: true, send_author: name, send_title: title, send_text: messageText},
 86         function (response) {
 87             HabraCorrector.prototype.sendDataPost(response.send_data);
 88         });
 89 };
 90 
 91 /**
 92  * Заполняет необходимые поля и показывает диалоговое окно пользователю.
 93  */
 94 HabraCorrector.prototype.startDialog = function() {
 95     var habraCorrector = this;
 96     if (habraCorrector.habraPage.isCurrentUrlCorrect()) {
 97         var articleUrl = document.baseURI,
 98             author = habraCorrector.habraPage.getAuthorName(),
 99             title = 'Ошибка в статье "' + habraCorrector.habraPage.getArticleTitle() + '"',
100             content = habraCorrector.habraPage.getContentText(articleUrl);
101         habraCorrector.dialogBox.resetDialogFields();
102         habraCorrector.dialogBox.setDialogContent(author, title, content);
103         habraCorrector.dialogBox.showDialog(author, title, habraCorrector.sendRequestToBackgroundPage);
104     }
105 };
106 
107 /**
108  * Добавляет событие для сочетания клавиш Ctrl+Enter.
109  * При срабатывании показывает диалоговое окно для отправки сообщения об ошибке.
110  */
111 HabraCorrector.prototype.addHotKey = function() {
112     var habraCorrector = this;
113     window.addEventListener("keydown", function (event) {
114         var modifier = event.ctrlKey || event.metaKey;
115         if (modifier && event.keyCode === 13) {
116             habraCorrector.startDialog();
117         }
118     }, false);
119 };
