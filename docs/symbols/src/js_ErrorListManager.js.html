  7 /**
  8  * @class Отвечает за работу со списком ошибок.
  9  * @constructor
 10  * @name ErrorListManager
 11  * @param {object} errorDict JSON-представление списка ошибок
 12  */
 13 var ErrorListManager = function (errorDict) {
 14     'use strict';
 15     this.errorDict = errorDict;
 16 };
 17 
 18 /**
 19  * Возвращает информацию для сообщения, соответсвующую типу ошибки.
 20  *
 21  * @param {string} errorType Тип ошибки
 22  * @return {string} Текст сообщения об ошибке
 23  */
 24 ErrorListManager.prototype.getErrorMessageText = function (errorType) {
 25     'use strict';
 26     var errorMessageText,
 27         errorListManager = this;
 28 
 29     $.each(errorListManager.errorDict, function () {
 30         if (this.title === errorType) {
 31             errorMessageText = this.messageText;
 32             return false;
 33         }
 34         if (errorListManager.isErrorHasSubTypes(this.title)) {
 35             $.each(this.children, function () {
 36                 if (this.title === errorType) {
 37                     errorMessageText = this.messageText;
 38                     return false;
 39                 }
 40             });
 41         }
 42     });
 43     return errorMessageText;
 44 };
 45 
 46 
 47 /**
 48  * Есть ли у типа ошибки её подтипы.
 49  *
 50  * @param {string} errorType
 51  * @return {boolean} true, если у типа есть подтип, иначе false
 52  */
 53 ErrorListManager.prototype.isErrorHasSubTypes = function (errorType) {
 54     'use strict';
 55     var hasSubTypes = false,
 56         errorListManager = this;
 57 
 58     $.each(errorListManager.errorDict, function () {
 59         if (this.title === errorType && this.children !== undefined) {
 60             hasSubTypes = true;
 61             return false;
 62         }
 63     });
 64 
 65     return hasSubTypes;
 66 };
