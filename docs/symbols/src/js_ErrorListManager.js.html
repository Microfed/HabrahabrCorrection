  7 /**
  8  * @class Отвечает за работу со списком ошибок.
  9  * @constructor
 10  * @name ErrorListManager
 11  * @param {object} errorDict JSON-представление списка ошибок
 12  */
 13 var ErrorListManager = function(errorDict) {
 14     this.errorDict = errorDict;
 15 };
 16 
 17 /**
 18  * Возвращает информацию для сообщения, соответсвующую типу ошибки.
 19  *
 20  * @param {string} errorType Тип ошибки
 21  * @return {string} Текст сообщения об ошибке
 22  */
 23 ErrorListManager.prototype.getErrorMessageText = function (errorType) {
 24     var errorMessageText;
 25     var errorListManager = this;
 26 
 27     $.each(errorListManager.errorDict, function() {
 28         if (this.title === errorType) {
 29             errorMessageText = this.messageText;
 30             return false;
 31         }
 32         if (errorListManager.isErrorHasSubTypes(this.title)) {
 33             $.each(this.children, function() {
 34                 if (this.title === errorType) {
 35                     errorMessageText = this.messageText;
 36                     return false;
 37                 }
 38             });
 39         }
 40     });
 41     return errorMessageText;
 42 };
 43 
 44 
 45 /**
 46  * Есть ли у типа ошибки её подтипы.
 47  *
 48  * @param {string} errorType
 49  * @return {boolean} true, если у типа есть подтип, иначе false
 50  */
 51 ErrorListManager.prototype.isErrorHasSubTypes = function (errorType) {
 52     var hasSubTypes = false;
 53     var errorListManager = this;
 54 
 55     $.each(errorListManager.errorDict, function() {
 56         if (this.title === errorType && this.children !== undefined) {
 57             hasSubTypes = true;
 58             return false;
 59         }
 60     });
 61 
 62     return hasSubTypes;
 63 };
