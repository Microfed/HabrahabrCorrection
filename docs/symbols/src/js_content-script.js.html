 24     /**
 25      * Проверяет, подходит ли текущая страница для работы скрипта.
 26      *
 27      * @return {boolean} Возвращает true, если текущая страница - это страница на хабре
 28      * с комментами (это означает, что открыта страница статьи).
 29      */
 30     function isCurrentUrlCorrect() {
 31         var isCorrect = false,
 32             url = document.baseURI;
 33 
 34         if (url.search(HABRAHABR_URL) !== -1) {
 35             isCorrect = true;
 36         }
 37         if (document.getElementById("comments") !== undefined) {
 38             isCorrect = isCorrect && true;
 39         } else {
 40             isCorrect = false;
 41         }
 42         return isCorrect;
 43     }

 99     /**
100      * Возвращает тему в разделе правил.
101      *

103      * @return {string} часть ссылки на правило, соответсвующее выбранной теме.

105     function getErrorSubType(errorSubType) {
106         var topic = '';
107         switch (errorSubType) {
108         case "5_2":
109             topic = '-nouns';
110             break;
111         case "5_3":

114         case "5_4":
115             topic = '-numerals';

117         case "5_5":
118             topic = '-adverbs';
119             break;
120         case "5_6":
121             topic = '-prepositions';
122             break;
123         case "5_7":
124             topic = '-particles';
125             break;
126         }
127         return topic;
128     }

141     /**
142      * Добавляет событие onChange для элемента c id='type-of-error' диалогового окна.
143      * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
144      * зависят от его содержания.
145      */
146     function addEventTypeOfError_Change() {
147         $('#type-of-error').change(function () {
148             var errorType = getErrorType(),
149                 subtypeOfErrorSelect = $('#subtype-of-error'),
150                 listToAppend = '',
151                 messageToAppend = '\n';
152 
153             subtypeOfErrorSelect.empty();
154 
155             switch (errorType) {
156             case "1":
157                 messageToAppend += 'Пропущен пробел.';
158                 break;
159             case "2":
160                 messageToAppend += 'Лишняя запятая.';
161                 break;
162             case "3":
163                 messageToAppend += '<a href="http://tsya.ru">tsya.ru</a>';
164                 break;
165             case "4":
166                 listToAppend = getCommasList();
167                 break;
168             case "5":
169                 listToAppend = getListOfSpellings();
170                 break;
171             }
172 
173             if (listToAppend.length > 1) {
174                 subtypeOfErrorSelect.append(listToAppend);
175                 subtypeOfErrorSelect.show();
176             }
177 
178             addTextToMessage(messageToAppend);
179         });
180     }

182     /**
183      * Добавляет событие onChange для элемента c id='subtype-of-error' диалогового окна.
184      * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
185      * зависят от его содержания.
186      */
187     function addEventSubTypeOfError_Change() {
188         $('#subtype-of-error').change(function () {
189             var messageToAppend = '\n',
190                 topic = '',
191                 errorSubType = $('#subtype-of-error').val();
192 
193             if (errorSubType[0] === '4') {
194                 messageToAppend += THERULES_URL + 'comma-' + errorSubType.substr(2) + '/">' + TYPE_OF_ERROR_COMMA + '</a>';
195             } else {
196                 topic = getErrorSubType(errorSubType);
197                 messageToAppend += THERULES_URL + 'hyphen' + topic + '/">' + TYPE_OF_ERROR_SPELLING + '</a>';
198             }
199 
200             addTextToMessage(messageToAppend);
201         });
202     }

253      * @return {string} Название статьи

255     function getArticleTitle() {
256         var title = $('.post .title');
257         //удаляем все лишние символы в начале и конце заголовка

261     /**
262      * Возвращает выделенный текст.
263      *
264      * @return {string} Выделенный текст
265      */
266     function getSelectedText() {
267         var txt;
268         if (window.getSelection) {
269             txt = window.getSelection().toString();
270         } else if (document.getSelection) {
271             txt = document.getSelection();
272         } else if (document.selection) {
273             txt = document.selection.createRange().text;
274         }
275         return txt;
276     }

350     /**
351      * Проверяет ответ севера. Если ответ содержит информацию
352      * об ошибке, сообщает эту информацию пользователю
353      * посредством alert()
354      *
355      * @param response Ответ на запрос к серверу
356      */
357     function alertWhenFail(response) {
358         var resp = $(response).find('error');
359         if (resp[0] !== undefined) {
360             alert(resp[0].textContent);
361         }
362     }

364     /**
365      * Отправляет данные на сервер post-методом
366      *
367      * @param data Данные
368      * @param url Адрес отправки
369      */
370     function sendDataPost(data, url) {
371         $.post(url, data, function (response) {
372             alertWhenFail(response);
373         });

376     /**
377      * Отправляет сообщение фоновой странице расширения, отправляя данные для формирования сообщения
378      * об ошибке; принимает ответ и шлет сформированное сообщение автору статьи. Для успешнйо отправки
379      * сообщения требуется быть аутентифицированным пользователем.
380      *
381      * @param name Имя автора статьи
382      * @param title Название статьи
383      * @param messageText Текст сообщения
384      */
385     function sendRequestToBackgroundPage(name, title, messageText) {
386         chrome.extension.sendRequest({to_send_data: true, send_author: name, send_title: title, send_text: messageText},
387             function (response) {
388                 sendDataPost(response.send_data, URL_SEND_MESSAGE);
389             });
390     }

392     /**
393      * Показывает диалоговое окно для отправки сообщения автору статьи.
394      * Добавляет события для контролов диалогового окна.
395      *
396      * @param {string} name Имя автора статьи
397      * @param {string} title Название статьи
398      */
399     function showDialog(name, title) {
400         $('#habracorrect-dialog').attr('hidden', 'true');
401         $('#habracorrect-dialog').dialog({
402             width: 500
403         }, {
404             height: 360
405         }, {
406             buttons: {
407                 "Опечатка!": function () {
408                     addTextToMessage("\nОпечатка!");
409                 },
410                 "Отправить": function () {
411                     sendRequestToBackgroundPage(name, title, getMessageText());
412                     $(this).dialog("close");
413                 },
414                 "Отменить": function () {
415                     $(this).dialog("close");
416                 }
417             }
418         });
419 
420         $('.ui-dialog-buttonpane').find('button:contains("Опечатка!")').css("margin-right", "180px");
421     }

423     /**
424      * Заполняет необходимые поля и показывает диалоговое окно пользователю.
425      */
426     function startDialog() {
427         if (isCurrentUrlCorrect()) {
428             var articleUrl = document.baseURI,
429                 author = getAuthorName(),
430                 title = 'Ошибка в статье "' + getArticleTitle() + '"',
431                 content = getContentText(articleUrl);
432             resetDialogFields();
433             setDialogContent(author, title, content);
434             showDialog(author, title);
435         }
436     }

438     /**
439      * Добавляет событие для сочетания клавиш Ctrl+Enter.
440      * При срабатывании показывает диалоговое окно для отправки сообщения об ошибке.
441      */
442     window.addEventListener("keydown", function (event) {
443         var modifier = event.ctrlKey || event.metaKey;
444         if (modifier && event.keyCode === 13) {
