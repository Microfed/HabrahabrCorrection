 23     /**
 24      * Проверяет, подходит ли текущая страница для работы скрипта.
 25      *
 26      * @return {boolean} Возвращает true, если текущая страница - это страница на хабре
 27      * с комментами (это означает, что открыта страница статьи).
 28      */
 29     function isCurrentUrlCorrect() {
 30         var isCorrect = false,
 31             url = document.baseURI;
 32 
 33         if (url.search(HABRAHABR_URL) !== -1) {
 34             isCorrect = true;
 35         }
 36         if (document.getElementById("comments") !== undefined) {
 37             isCorrect = isCorrect && true;
 38         } else {
 39             isCorrect = false;
 40         }
 41         return isCorrect;
 42     }

 83     /**
 84      * Возвращает информацию для сообщения, соответсвующую типу ошибки.
 85      *
 86      * @param {string} errorType Тип ошибки
 87      * @return {string} Текст сообщения об ошибке
 88      */
 89     function getErrorMessageText(errorType) {
 90         var errorMessageText;
 91 
 92         $.each(error_dict.errorTypes, function() {
 93             if (this.optionText === errorType) {
 94                 errorMessageText = this.messageText;
 95                 return false;
 96             }
 97             if (isErrorHasSubTypes(this.optionText)) {
 98                 $.each(this.errorSubtypes, function() {
 99                     if (this.optionText === errorType) {
100                         errorMessageText = this.messageText;
101                         return false;
102                     }
103                 });
104             }
105         });
106         return errorMessageText;
107     }

109     /**
110      * Есть ли у типа ошибки её подтипы.
111      *
112      * @param {string} errorType
113      * @return {boolean} true, если у типа есть подтип, иначе false
114      */
115     function isErrorHasSubTypes(errorType) {
116         var hasSubTypes = false;
117 
118         $.each(error_dict.errorTypes, function() {
119             if (this.optionText === errorType && this.errorSubtypes !== undefined) {
120                 hasSubTypes = true;
121                 return false;
122             }
123         });
124 
125         return hasSubTypes;
126     }

128     /**
129      * Добавляет элементы option, которые соответствуют
130      * подтипам определенной ошибки, к заданному select
131      *
132      * @param {string} errorType Тип ошибки
133      * @param {jQuery object} subtypeOfErrorSelect Select для подтипов ошибок
134      */
135     function appendErrorSubTypesList(errorType, subtypeOfErrorSelect) {
136         $.each(error_dict.errorTypes, function() {
137             if (this.optionText === errorType) {
138                 $.each(this.errorSubtypes, function() {
139                     subtypeOfErrorSelect.append(new Option(this.optionText, 0));
140                 });
141             }
142         });
143     }

145     /**
146      * Добавляет событие onChange для элемента c id='type-of-error' диалогового окна.
147      * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
148      * зависят от его содержания.
149      */
150     function addEventTypeOfError_Change() {
151         $('#type-of-error').change(function () {
152             var errorType = getErrorType(),
153                 subtypeOfErrorSelect = $('#subtype-of-error'),
154                 messageToAppend = '\n';
155 
156             subtypeOfErrorSelect.empty();
157             messageToAppend += getErrorMessageText(errorType);
158 
159             if (isErrorHasSubTypes(errorType)) {
160                 appendErrorSubTypesList(errorType, subtypeOfErrorSelect);
161                 subtypeOfErrorSelect.show();
162             }
163 
164             addTextToMessage(messageToAppend);
165         });
166     }

168     /**
169      * Добавляет событие onChange для элемента c id='subtype-of-error' диалогового окна.
170      * При срабатывании события проверяется выделенный элемент. Дальнейшие действия
171      * зависят от его содержания.
172      */
173     function addEventSubTypeOfError_Change() {
174         $('#subtype-of-error').change(function () {
175             var messageToAppend = '',
176                 errorSubType = getErrorSubType();
177 
178             messageToAppend += getErrorMessageText(errorSubType);
179             addTextToMessage(messageToAppend);
180         });
181     }

240     function getAuthorName() {
241         var author;
242         if ($(AUTHOR_CLASS_NAME).length) {
243             author = $(AUTHOR_CLASS_NAME + ' a').text();
244         } else {
245             // Некоторые расширения для хабры меняют имя класса

247         }

251     /**
252      * Возвращает заголовок статьи.
253      *
254      * @return {string} Название статьи
255      */
256     function getArticleTitle() {
257         var title = $('.post .title');
258         //удаляем все лишние символы в начале и конце заголовка
259         return title.text().replace(/^\s*|\s*$|[\t\n]/g, '');
260     }

262     /**
263      * Возвращает выделенный текст.
264      *
265      * @return {string} Выделенный текст
266      */
267     function getSelectedText() {
268         var txt;
269         if (window.getSelection) {
270             txt = window.getSelection().toString();
271         } else if (document.getSelection) {
272             txt = document.getSelection();
273         } else if (document.selection) {
274             txt = document.selection.createRange().text;
275         }
276         return txt;
277     }

351     /**
352      * Проверяет ответ севера. Если ответ содержит информацию
353      * об ошибке, сообщает эту информацию пользователю
354      * посредством alert()
355      *
356      * @param response Ответ на запрос к серверу
357      */
358     function alertWhenFail(response) {
359         var resp = $(response).find('error');
360         if (resp[0] !== undefined) {
361             alert(resp[0].textContent);
362         }
363     }

367      *
368      * @param data Данные
369      * @param url Адрес отправки
370      */
371     function sendDataPost(data, url) {
372         $.post(url, data, function (response) {
373             alertWhenFail(response);

377     /**
378      * Отправляет сообщение фоновой странице расширения, отправляя данные для формирования сообщения
379      * об ошибке; принимает ответ и шлет сформированное сообщение автору статьи. Для успешнйо отправки
380      * сообщения требуется быть аутентифицированным пользователем.
381      *
382      * @param name Имя автора статьи
383      * @param title Название статьи
384      * @param messageText Текст сообщения
385      */
386     function sendRequestToBackgroundPage(name, title, messageText) {
387         chrome.extension.sendRequest({to_send_data: true, send_author: name, send_title: title, send_text: messageText},
388             function (response) {
389                 sendDataPost(response.send_data, URL_SEND_MESSAGE);
390             });
391     }

393     /**
394      * Показывает диалоговое окно для отправки сообщения автору статьи.
395      * Добавляет события для контролов диалогового окна.
396      *
397      * @param {string} name Имя автора статьи
398      * @param {string} title Название статьи
399      */
400     function showDialog(name, title) {
401         $('#habracorrect-dialog').attr('hidden', 'true');
402         $('#habracorrect-dialog').dialog({
403             width: 500
404         }, {
405             height: 360
406         }, {
407             buttons: {
408                 "Опечатка!": function () {
409                     addTextToMessage("\nОпечатка!");
410                 },
411                 "Отправить": function () {
412                     sendRequestToBackgroundPage(name, title, getMessageText());
413                     $(this).dialog("close");
414                 },
415                 "Отменить": function () {
416                     $(this).dialog("close");
417                 }
418             }
419         });
420 
421         $('.ui-dialog-buttonpane').find('button:contains("Опечатка!")').css("margin-right", "180px");
422     }

424     /**
425      * Заполняет необходимые поля и показывает диалоговое окно пользователю.
426      */
427     function startDialog() {
428         if (isCurrentUrlCorrect()) {
429             var articleUrl = document.baseURI,
430                 author = getAuthorName(),
431                 title = 'Ошибка в статье "' + getArticleTitle() + '"',
432                 content = getContentText(articleUrl);
433             resetDialogFields();
434             setDialogContent(author, title, content);
435             showDialog(author, title);
436         }
437     }

440      * Добавляет событие для сочетания клавиш Ctrl+Enter.
441      * При срабатывании показывает диалоговое окно для отправки сообщения об ошибке.
442      */
443     window.addEventListener("keydown", function (event) {
444         var modifier = event.ctrlKey || event.metaKey;
445         if (modifier && event.keyCode === 13) {
