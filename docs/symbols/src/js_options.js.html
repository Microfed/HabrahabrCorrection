  6 /**
  7  * @return Возвращает слой, содержащий dynatree-дерево.
  8  */
  9 var getTreeDiv = function () {
 10     'use strict';
 11     return $("#errorTree");
 12 };
 13 
 14 /**
 15  * Показывает информацию об элементе дерева.
 16  *
 17  * @param node Элемент дерева
 18  */
 19 function printNodeInfo(node) {
 20     'use strict';
 21     $('#activeTreeElement #title').val(node.data.title);
 22     $('#activeTreeElement #messageText').val(node.data.messageText);
 23     if (node.data.isFolder) {
 24         $('#activeTreeElement #isFolder').attr("checked", "checked");
 25     }
 26     $('#activeTreeElement').show();
 27 }
 28 
 29 /**
 30  * Очищает информационную панель.
 31  */
 32 function clearNodeInfo() {
 33     'use strict';
 34     $("#activeTreeElement #title").empty();
 35     $("#activeTreeElement #messageText").empty();
 36     $('#activeTreeElement #isFolder').removeAttr("checked");
 37     $('#activeTreeElement').hide();
 38 }
 39 
 40 /**
 41  * Показывает заданное сообщение с анимацией.
 42  *
 43  * @param messageText Текст сообщения
 44  */
 45 function printStatusMessage(messageText) {
 46     'use strict';
 47     $('#status').append(messageText);
 48     $('#status').show();
 49     setTimeout(function () {
 50         $('#status').effect('fade', 500, function () {
 51             $('#status').empty();
 52         });
 53     }, 2000);
 54 }

 56 /**
 57  * Сохраняет текущие значения контролов в качестве соответствующих опций.
 58  */
 59 function save_options() {
 60     'use strict';
 61     localStorage["isAdvtAttachToMessage"] = document.getElementById('isAdvtAttachToMessage').checked;
 62 
 63     var dict = getTreeDiv().dynatree("getTree").toDict();
 64     localStorage["errorList"] = JSON.stringify(dict.children);
 65 
 66     printStatusMessage("Сохранено");
 67 }

 69 /**
 70  * Применяет стили для кнопок.
 71  */
 72 var setButtonUI = function () {
 73     'use strict';
 74     $('button', '#activeTreeElement').button();
 75 };

 77 /**
 78  * Удаляет активный элемент из дерева.
 79  */
 80 var deleteActiveElement = function () {
 81     'use strict';
 82     getTreeDiv().dynatree("getActiveNode").remove();
 83 
 84     //Fix error: плагин после удаления элемента срет в стиль этого дива (display: none)
 85     $('#activeTreeElement').show();
 86 };
 87 
 88 /**
 89  * Применяет изменения для активного элемента.
 90  */
 91 var acceptChangeActiveElement = function () {
 92     'use strict';
 93     var node = getTreeDiv().dynatree("getActiveNode"),
 94         title = $('#activeTreeElement #title').val(),
 95         messageText = $('#activeTreeElement #messageText').val(),
 96         isFolder = $('#activeTreeElement #isFolder').val();
 97     if (!node) {
 98         return;
 99     }
100     node.fromDict({
101         title: title,
102         messageText: messageText,
103         isFolder: isFolder
104     });
105 };
106 
107 /**
108  * Добавляет новый элемент в дерево.
109  */
110 var addNewElement = function () {
111     'use strict';
112     var parentNode,
113         root = getTreeDiv().dynatree("getRoot"),
114         activeNode = getTreeDiv().dynatree("getActiveNode"),
115         title = $('#activeTreeElement #title').val(),
116         messageText = $('#activeTreeElement #messageText').val(),
117         isFolder = false;
118 
119     if (activeNode !== null) {
120         if (activeNode.data.isFolder) {
121             parentNode = activeNode;
122         } else {
123             parentNode = activeNode.getParent();
124         }
125     } else {
126         parentNode = root;
127         activeNode = root;
128     }
129 
130     if (parentNode === root && !(activeNode.data.isFolder)) {
131         isFolder = ($('#activeTreeElement #isFolder:checked').val() !== undefined);
132     }
133 
134     parentNode.addChild({
135         title: title,
136         messageText: messageText,
137         isFolder: isFolder
138     });
139 
140 };

142 /**
143  * Добавляет кнопки для управления деревом ошибок,
144  * добавляет соответствующие события для кнопок.
145  *
146  * Используется библиотеки jQuery.dynatree, jQuery UI.
147  */
148 function addButtonsToManageTree() {
149     'use strict';
150     setButtonUI();
151 
152     $('#delete', '#activeTreeElement').click(deleteActiveElement);
153     $('#accept', '#activeTreeElement').click(acceptChangeActiveElement);
154     $('#add', '#activeTreeElement').click(addNewElement);
155 }

157 /**
158  * Устанавливает значения для дерева, корректно его отображает.
159  */
160 function setTreeInfo() {
161     'use strict';
162     var treeDiv = getTreeDiv();
163 
164     if (treeDiv.text() !== '') {
165         treeDiv.dynatree("getTree").reload();
166     } else {
167         treeDiv.dynatree({
168             onActivate: function (node) {
169                 printNodeInfo(node);
170             },
171             onDeactivate: function () {
172                 clearNodeInfo();
173             },
174             children: JSON.parse(localStorage["errorList"])
175         });
176 
177         addButtonsToManageTree();
178     }
179 }
180 
181 /**
182  * Получает список ошибок из файла (в папке расширения),
183  * устанавливает этот список как текущий,
184  * обновляет информацию в дереве ошибок,
185  * печатает сообщение, что все хорошо.
186  */
187 function setErrorListFromFile() {
188     'use strict';
189     $.getJSON(chrome.extension.getURL('/errorList.json'),
190         function (data) {
191             localStorage["errorList"] = JSON.stringify(data);
192             setTreeInfo();
193             printStatusMessage("Список ошибок загружен");
194         })
195         .error(function () {
196             printStatusMessage("Проблема при загрузке файла. Список ошибок не обновлен.");
197         });
198 }

200 /**
201  * Восстанавливает настройки.
202  */
203 function restore_options() {
204     'use strict';
205     var isAdvtAttachToMessage = localStorage["isAdvtAttachToMessage"],
206         result = false;
207 
208     if (isAdvtAttachToMessage === 'true') {
209         result = true;
210     }
211     document.getElementById('isAdvtAttachToMessage').checked = result;
212 
213     if (localStorage["errorList"] === undefined) {
214         setErrorListFromFile();
215     }
216 
217     setTreeInfo();
218 }
