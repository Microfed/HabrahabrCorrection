 13 /**
 14  * Проверяет, поддерживается LocalStorage.
 15  *
 16  * @return {boolean} Возвращает true, если браузер поддерживает LocalStorage, иначе - false
 17  */
 18 Options.isLocalStorageEnable = function () {
 19     'use strict';
 20     return (window.hasOwnProperty('localStorage') && window.localStorage !== null);
 21 };
 22 
 23 /**
 24  * Устанавливает значение заданного элемента
 25  * в локальном хранилище.
 26  *
 27  * @param {string} name Имя параметра
 28  * @param value Значение параметра
 29  */
 30 Options.setLocalStorageItem = function (name, value) {
 31     'use strict';
 32     if (Options.isLocalStorageEnable()) {
 33         localStorage.setItem(name, value);
 34     }
 35 };
 36 
 37 /**
 38  * Получает значение параметра из
 39  * локального хранилища.
 40  *
 41  * @param {string} name Имя параметра
 42  */
 43 Options.getLocalStorageItem = function (name) {
 44     'use strict';
 45     if (Options.isLocalStorageEnable()) {
 46         return localStorage.getItem(name);
 47     } else {
 48         return undefined;
 49     }
 50 };
 51 
 52 /**
 53  * @return Возвращает слой, содержащий dynatree-дерево.
 54  */
 55 Options.getTreeDiv = function () {
 56     'use strict';
 57     return $("#errorTree");
 58 };

 60 /**
 61  * Показывает информацию об элементе дерева.
 62  *
 63  * @param node Элемент дерева
 64  */
 65 Options.printNodeInfo = function (node) {
 66     'use strict';
 67     $('#activeTreeElement #title').val(node.data.title);
 68     $('#activeTreeElement #messageText').val(node.data.messageText);
 69     if (node.data.isFolder) {
 70         $('#activeTreeElement #isFolder').attr("checked", "checked");
 71     }
 72     $('#activeTreeElement').show();
 73 };
 74 
 75 /**
 76  * Очищает информационную панель.
 77  */
 78 Options.clearNodeInfo = function () {
 79     'use strict';
 80     $("#activeTreeElement #title").empty();
 81     $("#activeTreeElement #messageText").empty();
 82     $('#activeTreeElement #isFolder').removeAttr("checked");
 83     $('#activeTreeElement').hide();
 84 };

 86 /**
 87  * Показывает заданное сообщение с анимацией.
 88  *
 89  * @param messageText Текст сообщения
 90  */
 91 Options.printStatusMessage = function (messageText) {
 92     'use strict';
 93     $('#status').append(messageText);
 94     $('#status').show();
 95     setTimeout(function () {
 96         $('#status').effect('fade', 500, function () {
 97             $('#status').empty();
 98         });
 99     }, 2000);
100 };

102 /**
103  * Сохраняет текущие значения контролов в качестве соответствующих опций.
104  */
105 Options.save_options = function () {
106     'use strict';
107     Options.setLocalStorageItem('isAdvtAttachToMessage', document.getElementById('isAdvtAttachToMessage').checked);
108     Options.setLocalStorageItem('silentMode', document.getElementById('isSilentModeOn').checked);
109 
110     var dict = Options.getTreeDiv().dynatree("getTree").toDict();
111     Options.setLocalStorageItem('errorList', JSON.stringify(dict.children));
112 
113     Options.printStatusMessage("Сохранено");
114 };

124 /**
125  * Удаляет активный элемент из дерева.
126  */
127 Options.deleteActiveElement = function () {
128     'use strict';
129     Options.getTreeDiv().dynatree("getActiveNode").remove();
130 
131     //Fix error: плагин после удаления элемента срет в стиль этого дива (display: none)
132     $('#activeTreeElement').show();
133 };
134 
135 /**
136  * Применяет изменения для активного элемента.
137  */
138 Options.acceptChangeActiveElement = function () {
139     'use strict';
140     var node = Options.getTreeDiv().dynatree("getActiveNode"),
141         title = $('#activeTreeElement #title').val(),
142         messageText = $('#activeTreeElement #messageText').val(),
143         isFolder = $('input#isFolder').attr('checked') ? true : false;
144     console.log(isFolder);
145     if (!node) {
146         return;
147     }
148     node.fromDict({
149         title: title,
150         messageText: messageText,
151         isFolder: isFolder
152     });
153 };
154 
155 /**
156  * Добавляет новый элемент в дерево.
157  */
158 Options.addNewElement = function () {
159     'use strict';
160     var parentNode,
161         root = Options.getTreeDiv().dynatree("getRoot"),
162         activeNode = Options.getTreeDiv().dynatree("getActiveNode"),
163         title = $('#activeTreeElement #title').val(),
164         messageText = $('#activeTreeElement #messageText').val(),
165         isFolder = false;
166 
167     if (activeNode !== null) {
168         if (activeNode.data.isFolder) {
169             parentNode = activeNode;
170         } else {
171             parentNode = activeNode.getParent();
172         }
173     } else {
174         parentNode = root;
175         activeNode = root;
176     }
177 
178     if (parentNode === root && !(activeNode.data.isFolder)) {
179         isFolder = ($('#activeTreeElement #isFolder:checked').val() !== undefined);
180     }
181 
182     parentNode.addChild({
183         title: title,
184         messageText: messageText,
185         isFolder: isFolder
186     });
187 
188 };

192  * добавляет соответствующие события для кнопок.

194  * Используется библиотеки jQuery.dynatree, jQuery UI.

202     $('#add', '#activeTreeElement').click(Options.addNewElement);

205 /**
206  * Устанавливает значения для дерева, корректно его отображает.
207  */
208 Options.setTreeInfo = function () {
209     'use strict';
210     var treeDiv = Options.getTreeDiv();
211 
212     if (treeDiv.text() !== '') {
213         treeDiv.dynatree("getTree").reload();
214     } else {
215         treeDiv.dynatree({
216             onActivate: function (node) {
217                 Options.printNodeInfo(node);
218             },
219             onDeactivate: function () {
220                 Options.clearNodeInfo();
221             },
222             children: JSON.parse(Options.getLocalStorageItem('errorList'))
223         });
224 
225         Options.addButtonsToManageTree();
226     }
227 };

229 /**
230  * Получает список ошибок из файла (в папке расширения),
231  * устанавливает этот список как текущий,
232  * обновляет информацию в дереве ошибок,
233  * печатает сообщение, что все хорошо.
234  */
235 Options.setErrorListFromFile = function () {
236     'use strict';
237     $.getJSON(chrome.extension.getURL('/errorList.json'),
238         function (data) {
239             Options.setLocalStorageItem('errorList', JSON.stringify(data));
240             Options.setTreeInfo();
241             Options.printStatusMessage("Список ошибок загружен");
242         })
243         .error(function () {
244             Options.printStatusMessage("Проблема при загрузке файла. Список ошибок не обновлен.");
245         });
246 };

248 /**
249  * Возвращает соответствующее булево значение, содержащееся в строке.
250  * @param {string} string Строка для разбора
251  * @return {boolean} True, если строка содержит 'true', иначе False
252  */
253 Options.getBooleanValueFromString = function (string) {
254     'use strict';
255     var result = false;
256     if (string === 'true') {
257         result = true;
258     }
259     return result;
260 };

262 /**
263  * Восстанавливает настройки.
264  */
265 Options.restore_options = function () {
266     'use strict';
267     var isAdvtAttachToMessage = Options.getLocalStorageItem('isAdvtAttachToMessage'),
268         isSilentModeOn = Options.getLocalStorageItem('silentMode');
269 
270     if (isSilentModeOn === undefined) {
271         isSilentModeOn = 'false';
272         Options.setLocalStorageItem('silentMode', isSilentModeOn);
273     }
274 
275     document.getElementById('isAdvtAttachToMessage').checked = Options.getBooleanValueFromString(isAdvtAttachToMessage);
276     document.getElementById('isSilentModeOn').checked = Options.getBooleanValueFromString(isSilentModeOn);
277 
278     if (Options.getLocalStorageItem('errorList') === undefined) {
279         Options.setErrorListFromFile();
280     }
281 
282     Options.setTreeInfo();
283 };
