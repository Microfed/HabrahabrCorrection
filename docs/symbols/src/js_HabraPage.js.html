  6 /**
  7  * @class Отвечает за работу со страницей хабра-статьи.
  8  * @constructor 
  9  * @name HabraPage
 10  */
 11 var HabraPage = function() {
 12     this.AUTHOR_CLASS_NAME = '.author';
 13     this.AUTHOR_CLASS_NAME_MODIFIED = '.author.karmaloaded';
 14     this.HABRAHABR_URL = 'habrahabr.ru';
 15 };
 16 
 17 /**
 18  * Проверяет, подходит ли текущая страница для работы скрипта.
 19  *
 20  * @return {boolean} Возвращает true, если текущая страница - это страница на хабре
 21  * с комментами (это означает, что открыта страница статьи).
 22  */
 23 HabraPage.prototype.isCurrentUrlCorrect = function () {
 24     var isCorrect = false,
 25         url = document.baseURI;
 26 
 27     if (url.search(this.HABRAHABR_URL) !== -1) {
 28         isCorrect = true;
 29     }
 30     if (document.getElementById("comments") !== undefined) {
 31         isCorrect = isCorrect && true;
 32     } else {
 33         isCorrect = false;
 34     }
 35     return isCorrect;
 36 };
 37 
 38 /**
 39  * Возвращает имя автора статьи.
 40  *
 41  * @return {string} Ник автора статьи
 42  */
 43 HabraPage.prototype.getAuthorName = function () {
 44     var author;
 45     if ($(this.AUTHOR_CLASS_NAME).length) {
 46         author = $(this.AUTHOR_CLASS_NAME + ' a').text();
 47     } else {
 48         // Некоторые расширения для хабры меняют имя класса
 49         author = $(this.AUTHOR_CLASS_NAME_MODIFIED + ' a').text();
 50     }
 51     return author;
 52 };
 53 
 54 /**
 55  * Возвращает заголовок статьи.
 56  *
 57  * @return {string} Название статьи
 58  */
 59 HabraPage.prototype.getArticleTitle = function () {
 60     var title = $('.post .title');
 61     //удаляем все лишние символы в начале и конце заголовка
 62     return title.text().replace(/^\s*|\s*$|[\t\n]/g, '');
 63 };
 64 
 65 /**
 66  * Возвращает выделенный текст.
 67  *
 68  * @return {string} Выделенный текст
 69  */
 70 HabraPage.prototype.getSelectedText = function () {
 71     var txt;
 72     if (window.getSelection) {
 73         txt = window.getSelection().toString();
 74     } else if (document.getSelection) {
 75         txt = document.getSelection();
 76     } else if (document.selection) {
 77         txt = document.selection.createRange().text;
 78     }
 79     return txt;
 80 };
 81 
 82 /**
 83  *  Возвращает выделенный текст, как цитату.
 84  *
 85  * @param {string} articleUrl Ссылка на статью
 86  * @return {string} Текст ошибки, подготовленный для сообщения
 87  */
 88 HabraPage.prototype.getContentText = function (articleUrl) {
 89     return 'Ошибка в статье <a href="' + articleUrl + '">' + this.getArticleTitle() + '</a>\n<blockquote>' + this.getSelectedText() + '</blockquote>';
 90 };
