  6 /**
  7  * @class Отвечает за работу со страницей хабра-статьи.
  8  * @constructor
  9  * @name HabraPage
 10  */
 11 var HabraPage = function () {
 12     'use strict';
 13     this.AUTHOR_CLASS_NAME = 'div.vcard.author.full a.fn.nickname.url';
 14     this.AUTHOR_CLASS_NAME_MODIFIED = 'div.vcard.author.full a.fn.nickname.url.karmaloaded';
 15     this.HABRAHABR_URL = 'habrahabr.ru';
 16     this.selectedErrors = []; //текущий список выделенных пользователем ошибок
 17 };
 18 
 19 /**
 20  * Проверяет, подходит ли текущая страница для работы скрипта.
 21  *
 22  * @return {boolean} Возвращает true, если текущая страница - это страница на хабре
 23  * с комментами (это означает, что открыта страница статьи).
 24  */
 25 HabraPage.prototype.isCurrentUrlCorrect = function () {
 26     'use strict';
 27     var isCorrect = false,
 28         url = document.baseURI;
 29 
 30     if (url.search(this.HABRAHABR_URL) !== -1) {
 31         isCorrect = true;
 32     }
 33     if (document.getElementById("comments") !== undefined) {
 34         isCorrect = isCorrect && true;
 35     } else {
 36         isCorrect = false;
 37     }
 38     return isCorrect;
 39 };
 40 
 41 /**
 42  * Возвращает имя автора статьи.
 43  *
 44  * @return {string} Ник автора статьи
 45  */
 46 HabraPage.prototype.getAuthorName = function () {
 47     'use strict';
 48     var author,
 49         length = $(this.AUTHOR_CLASS_NAME).length,
 50         i;
 51 
 52     if (length > 0) {
 53         for (i = 0; i < length; i = +1) {
 54             if ($(this.AUTHOR_CLASS_NAME).attr('title') === 'Автор текста') {
 55                 author = $(this.AUTHOR_CLASS_NAME).text();
 56             }
 57         }
 58     } else if ($(this.AUTHOR_CLASS_NAME_MODIFIED).length > 0) {
 59         // Некоторые расширения для хабры меняют имя класса
 60         author = $(this.AUTHOR_CLASS_NAME_MODIFIED).text();
 61     } else if ($('div.infopanel div.author a').length > 0) {
 62         author = $('div.infopanel div.author a').text();
 63     }
 64     return author;
 65 };
 66 
 67 /**
 68  * Возвращает заголовок статьи.
 69  *
 70  * @return {string} Название статьи
 71  */
 72 HabraPage.prototype.getArticleTitle = function () {
 73     'use strict';
 74     var title;
 75     if ($('div.hentry h2.entry-title.single-entry-title span.topic').length > 0) {
 76         title = $('div.hentry h2.entry-title.single-entry-title span.topic');
 77     } else if ($('div.post h2.title span.post_title').length > 0) {
 78         title = $('div.post h2.title span.post_title');
 79     }
 80 
 81     return title.text();
 82 };
 83 
 84 /**
 85  * Возвращает выделенный текст.
 86  *
 87  * @return {string} Выделенный текст
 88  */
 89 HabraPage.prototype.getSelectedText = function () {
 90     'use strict';
 91     var txt = '';
 92     if (window.getSelection) {
 93         txt = window.getSelection().toString();
 94     } else if (document.getSelection) {
 95         txt = document.getSelection();
 96     } else if (document.selection) {
 97         txt = document.selection.createRange().text;
 98     }
 99     return txt;
100 };
101 
102 /**
103  * @return {string} Возвращает строку, содержащую все выделенные пользователем
104  * ошибки с момента последней отправки сообщения об ошибке.
105  */
106 HabraPage.prototype.getErrorsText = function () {
107     'use strict';
108     var i,
109         textString = '',
110         length = this.selectedErrors.length;
111 
112     for (i = 0; i < length; i += 1) {
113         textString += this.selectedErrors[i];
114     }
115 
116     return textString;
117 };
118 
119 /**
120  *  Возвращает выделенный текст, как цитату.
121  *
122  * @param {string} articleUrl Ссылка на статью
123  * @return {string} Текст ошибки, подготовленный для сообщения
124  */
125 HabraPage.prototype.getContentText = function (articleUrl) {
126     'use strict';
127     return 'Ошибка в статье <a href="' + articleUrl + '">' + this.getArticleTitle() + '</a>' + this.getErrorsText();
128 };
129 
130 /**
131  * Добавляет выделенный текст в буфер для последующего включения
132  * текста в сообщение об ошибках.
133  */
134 HabraPage.prototype.addErrorText = function () {
135     'use strict';
136     var selectedText = this.getSelectedText();
137     if (selectedText !== '') {
138         this.selectedErrors.push('\n<blockquote>' + selectedText + '</blockquote>');
139     }
140 };
141 
142 /**
143  * Очищает буфер выделенных ошибок. Обычно срабатывает после
144  * отправки сообщения об ошибках пользователю.
145  */
146 HabraPage.prototype.resetState = function () {
147     'use strict';
148     this.selectedErrors = [];
149 };
