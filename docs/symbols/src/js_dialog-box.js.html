 15 /**
 16  * Загружает и добавляет файл стилей jQuery UI для
 17  * темы Redmond.
 18  */
 19 DialogBox.prototype.appendStyle = function() {
 20     $('head').append('<link type="text/css" href="' +
 21         chrome.extension.getURL("css/redmond/jquery-ui-1.8.16.custom.css") +
 22         '" rel="stylesheet" />');
 23 }

 25 /**
 26  * Добавляет текст к элементу textarea диалогового окна.
 27  *
 28  * @param {string} text Текст для добавления
 29  */
 30 DialogBox.prototype.addTextToMessage = function (text) {
 31     var messageTextarea = $('#dialog-message-text');
 32     messageTextarea.text(messageTextarea.val(messageTextarea.val() + text));
 33 }
